1//! SuperInstance Privacy - Redaction & Token Vault
2//!
3//! This crate handles all privacy-related functionality:
4//! - Pattern detection (emails, phones, API keys, etc.)
5//! - Redaction with reversible tokens
6//! - Secure token vault for storing original values
7//! - Reinflation of responses
8//!
9//! # Privacy Flow
10//!
11//! ```text
12//! User Input → Detect Patterns → Redact → [TOKEN_XXXX] → Cloud
13//!                                              ↓
14//!                               Store in Vault (local only)
15//!                                              ↓
16//! Response ← Reinflate ← [TOKEN_XXXX] ← Cloud Response
17//! ```
18
19pub mod patterns;
20pub mod redactor;
21pub mod vault;
22
23// Re-exports
24pub use patterns::{Pattern, PatternMatch, PatternSet, PatternType};
25pub use redactor::{RedactionResult, Redactor, RedactorConfig};
26pub use vault::{SessionStats, TokenVault};
27
28/// Result type for privacy operations
29pub type PrivacyResult<T> = std::result::Result<T, PrivacyError>;
30
31/// Privacy error types
32#[derive(Debug, thiserror::Error)]
33pub enum PrivacyError {
34    #[error("Pattern compilation failed: {0}")]
35    PatternError(String),
36
37    #[error("Vault error: {0}")]
38    VaultError(String),
39
40    #[error("Token not found: {0}")]
41    TokenNotFound(String),
42
43    #[error("Database error: {0}")]
44    DatabaseError(#[from] rusqlite::Error),
45
46    #[error("Internal error: {0}")]
47    Internal(String),
48}
49
50/// Statistics about redaction
51#[derive(Debug, Clone, Default, serde::Serialize, serde::Deserialize)]
52pub struct RedactionStats {
53    /// Total patterns detected
54    pub patterns_detected: usize,
55    /// Patterns redacted
56    pub patterns_redacted: usize,
57    /// Tokens created
58    pub tokens_created: usize,
59    /// By pattern type
60    pub by_type: std::collections::HashMap<String, usize>,
61}
