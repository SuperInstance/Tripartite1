1//! Frame format for tunnel messages
2//!
3//! All messages sent over the QUIC tunnel use this frame format:
4//! +------+----------------+------------------------+
5//! | Type | Length (4B, BE) | Payload (JSON)         |
6//! +------+----------------+------------------------+
7
8use crate::error::{CloudError, CloudResult};
9use crate::protocol::messages::TunnelMessage;
10
11/// Message type byte
12#[derive(Debug, Clone, Copy, PartialEq)]
13#[repr(u8)]
14pub enum FrameType {
15    /// Heartbeat from client
16    Heartbeat = 0x01,
17
18    /// Heartbeat acknowledgment
19    HeartbeatAck = 0x02,
20
21    /// Escalation request
22    EscalationRequest = 0x03,
23
24    /// Escalation response
25    EscalationResponse = 0x04,
26
27    /// Stream chunk
28    StreamChunk = 0x05,
29
30    /// Stream end
31    StreamEnd = 0x06,
32
33    /// Error
34    Error = 0x07,
35
36    /// Pre-warm signal
37    PrewarmSignal = 0x08,
38}
39
40impl FrameType {
41    /// Convert from byte
42    pub fn from_byte(b: u8) -> CloudResult<Self> {
43        match b {
44            0x01 => Ok(FrameType::Heartbeat),
45            0x02 => Ok(FrameType::HeartbeatAck),
46            0x03 => Ok(FrameType::EscalationRequest),
47            0x04 => Ok(FrameType::EscalationResponse),
48            0x05 => Ok(FrameType::StreamChunk),
49            0x06 => Ok(FrameType::StreamEnd),
50            0x07 => Ok(FrameType::Error),
51            0x08 => Ok(FrameType::PrewarmSignal),
52            _ => Err(CloudError::validation(format!("Invalid frame type: 0x{:02x}", b))),
53        }
54    }
55
56    /// Convert to byte
57    pub fn to_byte(self) -> u8 {
58        self as u8
59    }
60}
61
62/// Wire frame
63///
64/// Binary format: [Type: 1B][Length: 4BE][Payload: JSON]
65#[derive(Debug, Clone)]
66pub struct Frame {
67    /// Message type byte
68    pub frame_type: FrameType,
69    /// Serialized message payload (JSON)
70    pub payload: Vec<u8>,
71}
72
73impl Frame {
74    /// Maximum frame size (10MB)
75    pub const MAX_SIZE: usize = 10 * 1024 * 1024;
76
77    /// Create a new frame
78    pub fn new(frame_type: FrameType, payload: Vec<u8>) -> CloudResult<Self> {
79        if payload.len() > Self::MAX_SIZE {
80            return Err(CloudError::validation(format!(
81                "Frame too large: {} bytes (max {})",
82                payload.len(),
83                Self::MAX_SIZE
84            )));
85        }
86
87        Ok(Self { frame_type, payload })
88    }
89
90    /// Create frame from message
91    pub fn from_message(message: TunnelMessage) -> CloudResult<Self> {
92        let frame_type = match &message {
93            TunnelMessage::Heartbeat(_) => FrameType::Heartbeat,
94            TunnelMessage::HeartbeatAck(_) => FrameType::HeartbeatAck,
95            TunnelMessage::EscalationRequest(_) => FrameType::EscalationRequest,
96            TunnelMessage::EscalationResponse(_) => FrameType::EscalationResponse,
97            TunnelMessage::StreamChunk(_) => FrameType::StreamChunk,
98            TunnelMessage::StreamEnd(_) => FrameType::StreamEnd,
99            TunnelMessage::Error(_) => FrameType::Error,
100            TunnelMessage::PrewarmSignal(_) => FrameType::PrewarmSignal,
101        };
102
103        let payload = serde_json::to_vec(&message)
104            .map_err(CloudError::Serialization)?;
105
106        Self::new(frame_type, payload)
107    }
108
109    /// Encode frame to bytes
110    pub fn encode(&self) -> Vec<u8> {
111        let mut bytes = Vec::with_capacity(5 + self.payload.len());
112
113        // Type byte
114        bytes.push(self.frame_type.to_byte());
115
116        // Length (4 bytes big-endian)
117        let len = self.payload.len() as u32;
118        bytes.extend_from_slice(&len.to_be_bytes());
119
120        // Payload
121        bytes.extend_from_slice(&self.payload);
122
123        bytes
124    }
125
126    /// Decode frame from bytes
127    pub fn decode(data: &[u8]) -> CloudResult<Self> {
128        if data.len() < 5 {
129            return Err(CloudError::validation("Frame too short (min 5 bytes)"));
130        }
131
132        let frame_type = FrameType::from_byte(data[0])?;
133
134        let len_bytes = [data[1], data[2], data[3], data[4]];
135        let payload_len = u32::from_be_bytes(len_bytes) as usize;
136
137        if data.len() < 5 + payload_len {
138            return Err(CloudError::validation(format!(
139                "Incomplete frame: expected {} bytes, got {}",
140                5 + payload_len,
141                data.len()
142            )));
143        }
144
145        let payload = data[5..5 + payload_len].to_vec();
146
147        Ok(Self { frame_type, payload })
148    }
149
150    /// Parse payload as message
151    pub fn to_message(&self) -> CloudResult<TunnelMessage> {
152        serde_json::from_slice(&self.payload)
153            .map_err(CloudError::Serialization)
154    }
155
156    /// Get payload length
157    pub fn payload_len(&self) -> usize {
158        self.payload.len()
159    }
160}

162#[cfg(test)]
163mod tests {
164    use super::*;
165    use crate::protocol::messages::*;
166
167    #[test]
168    fn test_frame_type_conversion() {
169        assert_eq!(FrameType::Heartbeat.to_byte(), 0x01);
170        assert_eq!(FrameType::from_byte(0x01).unwrap(), FrameType::Heartbeat);
171    }
172
173    #[test]
174    fn test_frame_encode_decode() {
175        let frame = Frame::new(
176            FrameType::Heartbeat,
177            b"test payload".to_vec(),
178        ).unwrap();
179
180        let encoded = frame.encode();
181        assert_eq!(encoded.len(), 5 + 12); // 5 header + 12 payload
182        assert_eq!(encoded[0], 0x01);
183
184        let decoded = Frame::decode(&encoded).unwrap();
185        assert_eq!(decoded.frame_type, FrameType::Heartbeat);
186        assert_eq!(decoded.payload, b"test payload");
187    }
188
189    #[test]
190    fn test_frame_too_large() {
191        let payload = vec![0u8; Frame::MAX_SIZE + 1];
192        let result = Frame::new(FrameType::Heartbeat, payload);
193        assert!(result.is_err());
194    }
195
196    #[test]
197    fn test_frame_from_message() {
198        let msg = TunnelMessage::Heartbeat(HeartbeatData {
199            device_id: "test".to_string(),
200            timestamp: 123456,
201            sequence: 1,
202            vitals: serde_json::json!({}),
203        });
204
205        let frame = Frame::from_message(msg).unwrap();
206        assert_eq!(frame.frame_type, FrameType::Heartbeat);
207
208        let encoded = frame.encode();
209        let decoded = Frame::decode(&encoded).unwrap();
210        let recovered_msg = decoded.to_message().unwrap();
211
212        assert!(matches!(recovered_msg, TunnelMessage::Heartbeat(_)));
213    }
214
215    #[test]
216    fn test_frame_invalid_type() {
217        let result = FrameType::from_byte(0xFF);
218        assert!(result.is_err());
219    }
220
221    #[test]
222    fn test_frame_too_short() {
223        let data = vec![0x01, 0x00, 0x00, 0x00]; // Only 4 bytes
224        let result = Frame::decode(&data);
225        assert!(result.is_err());
226    }
227
228    #[test]
229    fn test_frame_incomplete() {
230        let mut data = vec![0x01, 0x00, 0x00, 0x00, 0x0A]; // Says 10 bytes payload
231        data.extend_from_slice(&[0u8; 5]); // But only provides 5
232
233        let result = Frame::decode(&data);
234        assert!(result.is_err());
235    }
