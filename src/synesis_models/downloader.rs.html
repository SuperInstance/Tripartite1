1//! Model Downloader
2//!
3//! Downloads models from various sources (HuggingFace, direct URLs)
4//! with progress tracking, checksum verification, and resume support.
5
6use std::path::{Path, PathBuf};
7use std::sync::Arc;
8use tracing::{debug, info, instrument, warn};
9
10use crate::{ModelError, ModelResult};
11
12/// Download progress callback
13pub type ProgressCallback = Arc<dyn Fn(DownloadProgress) + Send + Sync>;
14
15/// Download progress information
16#[derive(Debug, Clone)]
17pub struct DownloadProgress {
18    /// Current bytes downloaded
19    pub downloaded: u64,
20    /// Total bytes (if known)
21    pub total: Option<u64>,
22    /// Current download speed in bytes/sec
23    pub speed: u64,
24    /// Estimated time remaining in seconds
25    pub eta: Option<u64>,
26    /// Current phase
27    pub phase: DownloadPhase,
28}
29
30#[derive(Debug, Clone, Copy, PartialEq, Eq)]
31pub enum DownloadPhase {
32    /// Checking if file exists / getting size
33    Checking,
34    /// Downloading file
35    Downloading,
36    /// Verifying checksum
37    Verifying,
38    /// Extracting (if compressed)
39    Extracting,
40    /// Complete
41    Complete,
42}
43
44/// Model source configuration
45#[derive(Debug, Clone)]
46pub enum ModelSource {
47    /// HuggingFace Hub
48    HuggingFace {
49        repo_id: String,
50        filename: String,
51        revision: Option<String>,
52    },
53    /// Direct URL
54    Url { url: String, filename: String },
55    /// Local file (copy/link)
56    Local { path: PathBuf },
57}
58
59impl std::fmt::Display for ModelSource {
60    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
61        match self {
62            ModelSource::HuggingFace {
63                repo_id, filename, ..
64            } => {
65                write!(f, "{}/{}", repo_id, filename)
66            },
67            ModelSource::Url { url, .. } => {
68                write!(f, "{}", url)
69            },
70            ModelSource::Local { path } => {
71                write!(f, "{}", path.display())
72            },
73        }
74    }
75}

77/// Model download specification
78#[derive(Debug, Clone)]
79pub struct DownloadSpec {
80    /// Source of the model
81    pub source: ModelSource,
82    /// Expected SHA256 checksum

90/// Model downloader
91pub struct Downloader {

93    models_dir: PathBuf,

100impl Downloader {
101    /// Create a new downloader
102    pub fn new(models_dir: PathBuf) -> Self {
103        Self {
104            models_dir,
105            client: reqwest::Client::builder()
106                .user_agent("synesis-models/0.1.0")
107                .build()
108                .expect("Failed to create HTTP client"),
109            hf_token: std::env::var("HF_TOKEN").ok(),
110        }
111    }

115        self.hf_token = Some(token);
116        self

119    /// Download a model
120    #[instrument(skip(self, progress_callback))]
121    pub async fn download(
122        &self,
123        spec: &DownloadSpec,
124        progress_callback: Option<ProgressCallback>,
125    ) -> ModelResult<PathBuf> {
126        info!("Starting model download");
127
128        let dest_path = self.get_dest_path(spec)?;
129
130        // Check if already exists and valid
131        if dest_path.exists() {
132            if let Some(expected) = &spec.sha256 {
133                if self.verify_checksum(&dest_path, expected).await? {
134                    info!("Model already exists and checksum matches");
135                    if let Some(cb) = &progress_callback {
136                        cb(DownloadProgress {
137                            downloaded: spec.size_bytes.unwrap_or(0),
138                            total: spec.size_bytes,
139                            speed: 0,
140                            eta: Some(0),
141                            phase: DownloadPhase::Complete,
142                        });
143                    }
144                    return Ok(dest_path);
145                }
146                warn!("Existing file has invalid checksum, re-downloading");
147            } else {
148                info!("Model already exists (no checksum to verify)");
149                return Ok(dest_path);
150            }
151        }
152
153        // Create parent directories
154        if let Some(parent) = dest_path.parent() {
155            tokio::fs::create_dir_all(parent).await?;
156        }
157
158        // Download based on source
159        match &spec.source {
160            ModelSource::HuggingFace {
161                repo_id,
162                filename,
163                revision,
164            } => {
165                self.download_from_huggingface(
166                    repo_id,
167                    filename,
168                    revision.as_deref(),
169                    &dest_path,
170                    progress_callback.clone(),
171                )
172                .await?;
173            },
174            ModelSource::Url { url, .. } => {
175                self.download_from_url(url, &dest_path, progress_callback.clone())
176                    .await?;
177            },
178            ModelSource::Local { path } => {
179                self.copy_local(path, &dest_path).await?;
180            },
181        }
182
183        // Verify checksum if provided
184        if let Some(expected) = &spec.sha256 {
185            if let Some(cb) = &progress_callback {
186                cb(DownloadProgress {
187                    downloaded: spec.size_bytes.unwrap_or(0),
188                    total: spec.size_bytes,
189                    speed: 0,
190                    eta: None,
191                    phase: DownloadPhase::Verifying,
192                });
193            }
194
195            if !self.verify_checksum(&dest_path, expected).await? {
196                tokio::fs::remove_file(&dest_path).await?;
197                return Err(ModelError::ChecksumMismatch {
198                    model: spec.source.to_string(),
199                    expected: expected.clone(),
200                    actual: "invalid".to_string(),
201                });
202            }
203        }
204
205        if let Some(cb) = &progress_callback {
206            cb(DownloadProgress {
207                downloaded: spec.size_bytes.unwrap_or(0),
208                total: spec.size_bytes,
209                speed: 0,
210                eta: Some(0),
211                phase: DownloadPhase::Complete,
212            });
213        }
214
215        Ok(dest_path)
216    }

218    /// Get destination path for a model
219    fn get_dest_path(&self, spec: &DownloadSpec) -> ModelResult<PathBuf> {
220        let filename = match &spec.source {
221            ModelSource::HuggingFace { filename, .. } => filename.clone(),
222            ModelSource::Url { filename, .. } => filename.clone(),
223            ModelSource::Local { path } => path
224                .file_name()
225                .map(|s| s.to_string_lossy().to_string())
226                .ok_or_else(|| ModelError::InvalidPath("No filename".to_string()))?,
227        };
228
229        Ok(self.models_dir.join(&filename))
230    }

232    /// Download from HuggingFace Hub
233    async fn download_from_huggingface(
234        &self,
235        repo_id: &str,
236        filename: &str,
237        revision: Option<&str>,
238        dest: &Path,
239        progress_callback: Option<ProgressCallback>,
240    ) -> ModelResult<()> {
241        let revision = revision.unwrap_or("main");
242        let url = format!(
243            "https://huggingface.co/{}/resolve/{}/{}",
244            repo_id, revision, filename
245        );
246
247        debug!("Downloading from HuggingFace: {}", url);
248        self.download_from_url(&url, dest, progress_callback).await
249    }

251    /// Download from a URL
252    async fn download_from_url(
253        &self,
254        url: &str,
255        dest: &Path,
256        progress_callback: Option<ProgressCallback>,
257    ) -> ModelResult<()> {
258        let mut request = self.client.get(url);
259
260        // Add HuggingFace token if available and it's a HF URL
261        if url.contains("huggingface.co") {
262            if let Some(token) = &self.hf_token {
263                request = request.header("Authorization", format!("Bearer {}", token));
264            }
265        }
266
267        // Check for partial download (resume support)
268        let temp_path = dest.with_extension("part");
269        let resume_from = if temp_path.exists() {
270            let metadata = tokio::fs::metadata(&temp_path).await?;
271            let size = metadata.len();
272            if size > 0 {
273                request = request.header("Range", format!("bytes={}-", size));
274                Some(size)
275            } else {
276                None
277            }
278        } else {
279            None
280        };
281
282        let response = request.send().await?;
283
284        if !response.status().is_success() && response.status().as_u16() != 206 {
285            return Err(ModelError::DownloadFailed(format!(
286                "HTTP {}: {}",
287                response.status(),
288                response.status().canonical_reason().unwrap_or("Unknown")
289            )));
290        }
291
292        let total_size = response
293            .content_length()
294            .map(|cl| resume_from.unwrap_or(0) + cl);
295
296        // Open file for writing
297        let mut file = if resume_from.is_some() {
298            tokio::fs::OpenOptions::new()
299                .append(true)
300                .open(&temp_path)
301                .await?
302        } else {
303            tokio::fs::File::create(&temp_path).await?
304        };
305
306        let mut downloaded = resume_from.unwrap_or(0);
307        let mut last_progress = std::time::Instant::now();
308        let mut last_downloaded = downloaded;
309
310        // Stream the response
311        use futures_util::StreamExt;
312        use tokio::io::AsyncWriteExt;
313
314        let mut stream = response.bytes_stream();
315        while let Some(chunk_result) = stream.next().await {
316            let chunk = chunk_result?;
317            file.write_all(&chunk).await?;
318            downloaded += chunk.len() as u64;
319
320            // Update progress every 100ms
321            let now = std::time::Instant::now();
322            if now.duration_since(last_progress).as_millis() >= 100 {
323                let elapsed = now.duration_since(last_progress).as_secs_f64();
324                let bytes_since = downloaded - last_downloaded;
325                let speed = (bytes_since as f64 / elapsed) as u64;
326
327                let eta = total_size.map(|total| {
328                    if speed > 0 {
329                        (total - downloaded) / speed
330                    } else {
331                        0
332                    }
333                });
334
335                if let Some(cb) = &progress_callback {
336                    cb(DownloadProgress {
337                        downloaded,
338                        total: total_size,
339                        speed,
340                        eta,
341                        phase: DownloadPhase::Downloading,
342                    });
343                }
344
345                last_progress = now;
346                last_downloaded = downloaded;
347            }
348        }
349
350        file.flush().await?;
351        drop(file);
352
353        // Move temp file to final destination
354        tokio::fs::rename(&temp_path, dest).await?;
355
356        Ok(())
357    }

365    /// Verify SHA256 checksum
366    async fn verify_checksum(&self, path: &Path, expected: &str) -> ModelResult<bool> {
367        use sha2::{Digest, Sha256};
368
369        let mut file = tokio::fs::File::open(path).await?;
370        let mut hasher = Sha256::new();
371
372        let mut buffer = vec![0u8; 1024 * 1024]; // 1MB buffer
373        loop {
374            use tokio::io::AsyncReadExt;
375            let n = file.read(&mut buffer).await?;
376            if n == 0 {
377                break;
378            }
379            hasher.update(&buffer[..n]);
380        }
381
382        let result = hex::encode(hasher.finalize());
383        Ok(result == expected.to_lowercase())
384    }

386    /// List downloaded models
387    pub async fn list_downloaded(&self) -> ModelResult<Vec<PathBuf>> {
388        let mut models = Vec::new();
389
390        let mut entries = tokio::fs::read_dir(&self.models_dir).await?;
391        while let Some(entry) = entries.next_entry().await? {
392            let path = entry.path();
393            if path.is_file() {
394                if let Some(ext) = path.extension() {
395                    if ext == "gguf" || ext == "bin" || ext == "safetensors" {
396                        models.push(path);
397                    }
398                }
399            }
400        }
401
402        Ok(models)
403    }

405    /// Delete a downloaded model
406    pub async fn delete(&self, model_name: &str) -> ModelResult<()> {
407        let path = self.models_dir.join(model_name);
408        if path.exists() {
409            tokio::fs::remove_file(&path).await?;
410            info!("Deleted model: {}", model_name);
411        }
412        Ok(())
413    }
414
415    /// Get the size of a downloaded model
416    pub async fn get_size(&self, model_name: &str) -> ModelResult<u64> {
417        let path = self.models_dir.join(model_name);
418        let metadata = tokio::fs::metadata(&path).await?;
419        Ok(metadata.len())
420    }
