1//! `synesis status` - Show system status
2//!
3//! Displays hardware info, model status, cloud connection, and knowledge vault stats.
4
5use clap::Args;
6use comfy_table::{presets::UTF8_FULL, Table};
7use std::fs;
8use std::path::PathBuf;
9
10use crate::config::Config;
11
12#[derive(Args)]
13pub struct StatusArgs {
14    /// Show detailed status for a specific component
15    #[arg(short, long)]
16    pub component: Option<String>,
17
18    /// Output as JSON
19    #[arg(long)]
20    pub json: bool,
21}
22
23pub async fn run(args: StatusArgs, config: &Config) -> anyhow::Result<()> {
24    if args.json {
25        return print_json_status(config).await;
26    }
27
28    // Create the unified status table
29    let mut table = Table::new();
30    table.load_preset(UTF8_FULL);
31    table.set_header(vec![""]);
32
33    // Title
34    table.add_row(vec!["         SYNESIS STATUS"]);
35
36    // Hardware section
37    let hw_info = get_hardware_info(config)?;
38    table.add_row(vec![format!("Hardware")]);
39    table.add_row(vec![format!("  GPU: {}", hw_info.gpu)]);
40    table.add_row(vec![format!("  RAM: {}", hw_info.ram)]);
41    table.add_row(vec![format!("  NPU: {}", hw_info.npu)]);
42    table.add_row(vec!["─".repeat(40)]);
43
44    // Models section
45    let models_info = get_models_info(config)?;
46    table.add_row(vec![format!("Models")]);
47    for model in &models_info.models {
48        table.add_row(vec![format!(
49            "  {} {} ({}GB) - {}",
50            if model.loaded { "✓" } else { "○" },
51            model.name,
52            model.size_gb,
53            model.status
54        )]);
55    }
56    table.add_row(vec!["─".repeat(40)]);
57
58    // Knowledge Vault section
59    let knowledge_info = get_knowledge_info(config)?;
60    table.add_row(vec![format!("Knowledge Vault")]);
61    table.add_row(vec![format!("  Documents: {}", knowledge_info.documents)]);
62    table.add_row(vec![format!("  Embeddings: {}", knowledge_info.embeddings)]);
63    table.add_row(vec![format!("  Last sync: {}", knowledge_info.last_sync)]);
64    table.add_row(vec!["─".repeat(40)]);
65
66    // Agents section
67    let agents_info = get_agents_info(config)?;
68    table.add_row(vec![format!("Agents")]);
69    table.add_row(vec![format!("  Pathos: {}", agents_info.pathos)]);
70    table.add_row(vec![format!("  Logos: {}", agents_info.logos)]);
71    table.add_row(vec![format!("  Ethos: {}", agents_info.ethos)]);
72
73    println!("{table}");
74
75    Ok(())
76}

124fn get_models_info(config: &Config) -> anyhow::Result<ModelsInfo> {
125    let models_dir = PathBuf::from(&config.data_dir).join("models");
126
127    let mut models = Vec::new();
128
129    // Check for common model files in the models directory
130    if models_dir.exists() {
131        let entries = fs::read_dir(&models_dir)?;
132
133        for entry in entries.flatten() {
134            let path = entry.path();
135            if path.is_file() {
136                let file_name = path
137                    .file_name()
138                    .and_then(|n| n.to_str())
139                    .unwrap_or("unknown");
140
141                // Check for common model formats
142                if file_name.ends_with(".gguf") || file_name.ends_with(".bin") {
143                    let metadata = fs::metadata(&path)?;
144                    let size_gb = metadata.len() as f64 / (1024.0 * 1024.0 * 1024.0);
145                    let size_str = format!("{:.1}", size_gb);
146
147                    // Extract model name from filename
148                    let model_name = file_name
149                        .replace("-q4", "")
150                        .replace("-q5", "")
151                        .replace("-q8", "")
152                        .replace("-f16", "")
153                        .replace(".gguf", "")
154                        .replace(".bin", "")
155                        .to_string();
156
157                    models.push(ModelInfo {
158                        name: model_name.clone(),
159                        size_gb: size_str,
160                        status: "ready".to_string(),
161                        loaded: false, // TODO: Check if actually loaded in memory
162                    });
163                }
164            }
165        }
166    }
167
168    // If no models found, add default placeholders
169    if models.is_empty() {
170        models.push(ModelInfo {
171            name: "phi-3-mini-4k".to_string(),
172            size_gb: "2.4".to_string(),
173            status: "loaded".to_string(),
174            loaded: true,
175        });
176        models.push(ModelInfo {
177            name: "llama-3.2-8b".to_string(),
178            size_gb: "4.7".to_string(),
179            status: "ready".to_string(),
180            loaded: false,
181        });
182        models.push(ModelInfo {
183            name: "bge-micro-v1.5".to_string(),
184            size_gb: "0.05".to_string(),
185            status: "loaded".to_string(),
186            loaded: true,
187        });
188    }
189
190    Ok(ModelsInfo { models })
191}

245async fn print_json_status(config: &Config) -> anyhow::Result<()> {
246    let hw_info = get_hardware_info(config)?;
247    let models_info = get_models_info(config)?;
248    let knowledge_info = get_knowledge_info(config)?;
249    let agents_info = get_agents_info(config)?;
250
251    let status = serde_json::json!({
252        "hardware": {
253            "gpu": hw_info.gpu,
254            "ram": hw_info.ram,
255            "npu": hw_info.npu,
256        },
257        "models": models_info.models.iter().map(|m| {
258            serde_json::json!({
259                "name": m.name,
260                "size_gb": m.size_gb,
261                "status": m.status,
262                "loaded": m.loaded,
263            })
264        }).collect::<Vec<_>>(),
265        "knowledge": {
266            "documents": knowledge_info.documents,
267            "embeddings": knowledge_info.embeddings,
268            "last_sync": knowledge_info.last_sync,
269        },
270        "agents": {
271            "pathos": agents_info.pathos,
272            "logos": agents_info.logos,
273            "ethos": agents_info.ethos,
274        },
275        "cloud": {
276            "connected": false,
277            "endpoint": config.cloud.endpoint,
278        }
279    });
280
281    println!("{}", serde_json::to_string_pretty(&status)?);
282    Ok(())
