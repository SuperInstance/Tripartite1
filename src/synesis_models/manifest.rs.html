1//! Hardware Manifests
2//!
3//! Defines hardware profiles and model recommendations based on
4//! detected hardware capabilities.

6use crate::{HardwareDetector, HardwareInfo, ModelError, ModelResult, Quantization};

11/// Hardware manifest - maps hardware capabilities to model recommendations
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct HardwareManifest {
14    /// Profile name
15    pub name: String,
16    /// Description
17    pub description: String,
18    /// Minimum RAM required (bytes)
19    pub min_ram_bytes: u64,
20    /// Minimum VRAM required (bytes) - 0 for CPU-only
21    pub min_vram_bytes: u64,
22    /// Recommended models for each agent
23    pub recommendations: AgentRecommendations,
24    /// GPU layers to offload (0 for CPU-only)
25    pub gpu_layers: u32,
26    /// Context size
27    pub context_size: u32,
28}
29
30/// Model recommendations for each agent
31#[derive(Debug, Clone, Serialize, Deserialize)]
32pub struct AgentRecommendations {
33    pub pathos: ModelRecommendation,
34    pub logos: ModelRecommendation,
35    pub ethos: ModelRecommendation,
36    pub embeddings: ModelRecommendation,
37}

39/// A specific model recommendation
40#[derive(Debug, Clone, Serialize, Deserialize)]
41pub struct ModelRecommendation {
42    /// Model name/ID
43    pub model: String,
44    /// Quantization level
45    pub quantization: Quantization,
46    /// HuggingFace repo ID
47    pub repo_id: String,
48    /// Filename to download
49    pub filename: String,
50    /// Expected size in bytes
51    pub size_bytes: u64,
52    /// SHA256 checksum (optional)
53    pub sha256: Option<String>,
54}
55
56impl HardwareManifest {
57    /// Check if hardware meets this manifest's requirements
58    pub fn is_compatible(&self, hardware: &HardwareInfo) -> bool {
59        if hardware.ram_bytes < self.min_ram_bytes {
60            return false;
61        }
62
63        if self.min_vram_bytes > 0 {
64            let total_vram: u64 = hardware.gpu.as_ref().map_or(0, |g| g.vram_bytes);
65            if total_vram < self.min_vram_bytes {
66                return false;
67            }
68        }
69
70        true
71    }
72
73    /// Get total download size for all models
74    pub fn total_download_size(&self) -> u64 {
75        self.recommendations.pathos.size_bytes
76            + self.recommendations.logos.size_bytes
77            + self.recommendations.ethos.size_bytes
78            + self.recommendations.embeddings.size_bytes
79    }
80
81    /// Load manifest from a JSON file
82    #[instrument]
83    pub fn load(path: &Path) -> ModelResult<Self> {
84        debug!("Loading manifest from: {}", path.display());
85
86        let content = std::fs::read_to_string(path)
87            .map_err(|e| ModelError::InvalidPath(format!("Failed to read manifest: {}", e)))?;
88
89        let manifest: HardwareManifest = serde_json::from_str(&content)
90            .map_err(|e| ModelError::Internal(format!("Failed to parse manifest JSON: {}", e)))?;
91
92        // Validate the loaded manifest
93        manifest.validate()?;
94
95        Ok(manifest)
96    }
97
98    /// Validate manifest structure and values
99    pub fn validate(&self) -> ModelResult<()> {
100        // Check required fields
101        if self.name.is_empty() {
102            return Err(ModelError::Internal(
103                "Manifest name cannot be empty".to_string(),
104            ));
105        }
106
107        if self.min_ram_bytes == 0 {
108            return Err(ModelError::Internal(
109                "min_ram_bytes must be > 0".to_string(),
110            ));
111        }
112
113        // Validate model recommendations
114        self.validate_recommendation(&self.recommendations.pathos, "pathos")?;
115        self.validate_recommendation(&self.recommendations.logos, "logos")?;
116        self.validate_recommendation(&self.recommendations.ethos, "ethos")?;
117        self.validate_recommendation(&self.recommendations.embeddings, "embeddings")?;
118
119        // Validate context size
120        if self.context_size == 0 || self.context_size > 131072 {
121            return Err(ModelError::Internal(
122                "context_size must be between 1 and 131072".to_string(),
123            ));
124        }
125
126        debug!("Manifest '{}' validated successfully", self.name);
127        Ok(())
128    }
129
130    /// Validate a single model recommendation
131    fn validate_recommendation(
132        &self,
133        rec: &ModelRecommendation,
134        agent_name: &str,
135    ) -> ModelResult<()> {
136        if rec.model.is_empty() {
137            return Err(ModelError::Internal(format!(
138                "{} model name cannot be empty",
139                agent_name
140            )));
141        }
142
143        if rec.repo_id.is_empty() {
144            return Err(ModelError::Internal(format!(
145                "{} repo_id cannot be empty",
146                agent_name
147            )));
148        }
149
150        if rec.filename.is_empty() {
151            return Err(ModelError::Internal(format!(
152                "{} filename cannot be empty",
153                agent_name
154            )));
155        }
156
157        // Size can be 0 for shared models (like ethos reusing pathos)
158        // But if it's not 0, it should be reasonable
159        if rec.size_bytes > 0 && rec.size_bytes < 1024 {
160            return Err(ModelError::Internal(format!(
161                "{} size_bytes seems too small (< 1KB)",
162                agent_name
163            )));
164        }
165
166        Ok(())
167    }

169    /// Detect hardware and load appropriate manifest
170    #[instrument]
171    pub fn detect_and_load() -> ModelResult<Self> {
172        info!("Detecting hardware and selecting appropriate manifest...");
173
174        // Detect hardware
175        let hardware = HardwareDetector::detect()?;
176
177        // Get manifests directory
178        let manifests_dir = Self::manifests_dir()?;
179
180        // If manifests directory exists, try to find a matching manifest
181        if manifests_dir.exists() {
182            debug!("Searching for manifests in: {}", manifests_dir.display());
183
184            // Try to find a specific manifest for this hardware
185            if let Some(manifest) = Self::find_matching_manifest(&manifests_dir, &hardware)? {
186                info!("Found matching manifest: {}", manifest.name);
187                return Ok(manifest);
188            }
189        } else {
190            debug!("Manifests directory does not exist, using built-in profiles");
191        }
192
193        // Fallback to built-in profiles
194        info!("No matching manifest found, using built-in profile selection");
195        let profile = profiles::select_for_hardware(&hardware);
196        info!("Selected profile: {}", profile.name);
197        Ok(profile)
198    }
199
200    /// Get the manifests directory path
201    fn manifests_dir() -> ModelResult<PathBuf> {
202        let home = dirs::home_dir()
203            .ok_or_else(|| ModelError::Internal("Could not find home directory".to_string()))?;
204
205        Ok(home.join(".synesis").join("manifests"))
206    }
207
208    /// Find a manifest that matches the detected hardware
209    fn find_matching_manifest(dir: &Path, hardware: &HardwareInfo) -> ModelResult<Option<Self>> {
210        let entries = std::fs::read_dir(dir).map_err(|e| {
211            ModelError::Internal(format!("Failed to read manifests directory: {}", e))
212        })?;
213
214        // Try to find manifests with specific hardware identifiers
215        let manifests: Vec<_> = entries
216            .filter_map(|entry| entry.ok())
217            .filter(|entry| entry.path().extension().is_some_and(|ext| ext == "json"))
218            .collect();
219
220        debug!("Found {} manifest files", manifests.len());
221
222        for entry in manifests {
223            let path = entry.path();
224
225            // Try to load this manifest
226            match Self::load(&path) {
227                Ok(manifest) => {
228                    debug!("Checking manifest: {}", manifest.name);
229
230                    // Check if it's compatible
231                    if manifest.is_compatible(hardware) {
232                        return Ok(Some(manifest));
233                    }
234                },
235                Err(e) => {
236                    debug!("Failed to load manifest {:?}: {}", path, e);
237                    continue;
238                },
239            }
240        }
241
242        Ok(None)
243    }

245    /// Save manifest to file
246    pub fn save(&self, path: &Path) -> ModelResult<()> {
247        // Validate before saving
248        self.validate()?;
249
250        // Create parent directory if needed
251        if let Some(parent) = path.parent() {
252            std::fs::create_dir_all(parent)
253                .map_err(|e| ModelError::Internal(format!("Failed to create directory: {}", e)))?;
254        }
255
256        // Serialize and save
257        let json = serde_json::to_string_pretty(self)
258            .map_err(|e| ModelError::Internal(format!("Failed to serialize manifest: {}", e)))?;
259
260        std::fs::write(path, json)
261            .map_err(|e| ModelError::Internal(format!("Failed to write manifest: {}", e)))?;
262
263        info!("Saved manifest to: {}", path.display());
264        Ok(())
265    }

267    /// Install this manifest to the manifests directory
268    pub fn install(&self, name: &str) -> ModelResult<PathBuf> {
269        let manifests_dir = Self::manifests_dir()?;
270
271        // Create directory if needed
272        std::fs::create_dir_all(&manifests_dir).map_err(|e| {
273            ModelError::Internal(format!("Failed to create manifests directory: {}", e))
274        })?;
275
276        let manifest_path = manifests_dir.join(format!("{}.json", name));
277
278        self.save(&manifest_path)?;
279
280        info!(
281            "Installed manifest '{}' to: {}",
282            name,
283            manifest_path.display()
284        );
285        Ok(manifest_path)
286    }

288    /// Get a summary of the manifest
289    pub fn summary(&self) -> String {
290        format!(
291            "Profile: {}\n  {}\n  RAM: {}+, VRAM: {}+\n  Models: {}, {}, {}, {}",
292            self.name,
293            self.description,
294            format_bytes(self.min_ram_bytes),
295            if self.min_vram_bytes > 0 {
296                format_bytes(self.min_vram_bytes)
297            } else {
298                "N/A".to_string()
299            },
300            self.recommendations.pathos.model,
301            self.recommendations.logos.model,
302            self.recommendations.ethos.model,
303            self.recommendations.embeddings.model,
304        )
305    }

308/// Format bytes as human-readable string
309fn format_bytes(bytes: u64) -> String {
310    const KB: u64 = 1024;
311    const MB: u64 = KB * 1024;
312    const GB: u64 = MB * 1024;
313    const TB: u64 = GB * 1024;
314
315    if bytes >= TB {
316        format!("{:.1} TB", bytes as f64 / TB as f64)
317    } else if bytes >= GB {
318        format!("{:.1} GB", bytes as f64 / GB as f64)
319    } else if bytes >= MB {
320        format!("{:.1} MB", bytes as f64 / MB as f64)
321    } else if bytes >= KB {
322        format!("{:.1} KB", bytes as f64 / KB as f64)
323    } else {
324        format!("{} B", bytes)
325    }
326}

573    /// Select best profile for given hardware
574    pub fn select_for_hardware(hardware: &HardwareInfo) -> HardwareManifest {
575        // Check from highest to lowest
576        let profiles = [ultra(), performance(), standard(), minimal()];
577
578        for profile in profiles {
579            if profile.is_compatible(hardware) {
580                return profile;
581            }
582        }
583
584        // Fallback to minimal
585        minimal()
586    }
