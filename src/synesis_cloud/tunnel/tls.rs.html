14/// Create TLS configuration for mTLS connection
15///
16/// # Arguments
17/// * `cert_path` - Path to device certificate (PEM format)
18/// * `key_path` - Path to device private key (PEM format)
19///
20/// # Returns
21/// * `ClientConfig` configured for mTLS with system root CAs
22pub fn create_tls_config(
23    cert_path: &Path,
24    key_path: &Path,
25) -> CloudResult<Arc<ClientConfig>> {
26    // Load device certificate
27    let cert_file = File::open(cert_path).map_err(|e| {
28        CloudError::certificate(format!("Failed to open certificate file: {}", e))
29    })?;
30    let mut cert_reader = BufReader::new(cert_file);
31    let cert_vec = certs(&mut cert_reader).map_err(|e| {
32        CloudError::certificate(format!("Failed to parse certificate: {}", e))
33    })?;
34    let certs = cert_vec.into_iter().map(Certificate).collect::<Vec<_>>();
35
36    if certs.is_empty() {
37        return Err(CloudError::certificate("No certificates found in file"));
38    }
39
40    // Load device private key (try RSA, then EC, then PKCS8)
41    let key_file = File::open(key_path).map_err(|e| {
42        CloudError::certificate(format!("Failed to open key file: {}", e))
43    })?;
44    let mut key_reader = BufReader::new(key_file);
45
46    // Try RSA private keys first
47    let keys = rsa_private_keys(&mut key_reader).map_err(|e| {
48        CloudError::certificate(format!("Failed to parse RSA key: {}", e))
49    })?;
50
51    let key = if !keys.is_empty() {
52        keys[0].clone()
53    } else {
54        // Reset reader and try EC keys
55        key_reader = BufReader::new(File::open(key_path).map_err(|e| {
56            CloudError::certificate(format!("Failed to reopen key file: {}", e))
57        })?);
58        let keys = ec_private_keys(&mut key_reader).map_err(|e| {
59            CloudError::certificate(format!("Failed to parse EC key: {}", e))
60        })?;
61
62        if !keys.is_empty() {
63            keys[0].clone()
64        } else {
65            // Reset reader and try PKCS8 keys
66            key_reader = BufReader::new(File::open(key_path).map_err(|e| {
67                CloudError::certificate(format!("Failed to reopen key file: {}", e))
68            })?);
69            let keys = pkcs8_private_keys(&mut key_reader).map_err(|e| {
70                CloudError::certificate(format!("Failed to parse PKCS8 key: {}", e))
71            })?;
72
73            if !keys.is_empty() {
74                keys[0].clone()
75            } else {
76                return Err(CloudError::certificate("No private key found in file"));
77            }
78        }
79    };
80
81    let key = PrivateKey(key);
82
83    // Build root certificate store with system CAs
84    let mut roots = RootCertStore::empty();
85    roots.add_trust_anchors(webpki_roots::TLS_SERVER_ROOTS.iter().map(|ta| {
86        rustls::OwnedTrustAnchor::from_subject_spki_name_constraints(
87            ta.subject,
88            ta.spki,
89            ta.name_constraints.as_ref().map(|nc| *nc),
90        )
91    }));
92
93    // Build client config with mTLS
94    let config = ClientConfig::builder()
95        .with_safe_defaults()
96        .with_root_certificates(roots)
97        .with_client_auth_cert(certs, key)
98        .map_err(|e| CloudError::tls(format!("Failed to build client config: {}", e)))?;
99
100    Ok(Arc::new(config))
101}

103/// Generate device certificate (called during `synesis init`)
104///
105/// This is a placeholder for future implementation.
106/// In production, certificates would be signed by the SuperInstance CA.
107///
108/// # Arguments
109/// * `device_id` - Unique device identifier
110///
111/// # Returns
112/// * `(Certificate, PrivateKey)` pair for the device
113pub fn generate_device_certificate(
114    device_id: &str,
115) -> CloudResult<(Certificate, PrivateKey)> {
116    use rcgen::{Certificate as RcgenCert, CertificateParams, DnType, KeyPair, SanType};
117
118    // Generate key pair
119    let key_pair = KeyPair::generate(&rcgen::PKCS_ECDSA_P256_SHA256)
120        .map_err(|e| CloudError::certificate(format!("Failed to generate key pair: {}", e)))?;
121
122    // Build certificate parameters
123    let mut params = CertificateParams::default();
124    params.distinguished_name.push(DnType::CommonName, format!("device-{}", device_id));
125    params.distinguished_name.push(DnType::OrganizationName, "SuperInstance");
126    params.not_before = time::OffsetDateTime::now_utc();
127    params.not_after = time::OffsetDateTime::now_utc() + time::Duration::days(365);
128    params.key_pair = Some(key_pair);
129
130    // Subject alternative name (DNS name for device)
131    params.subject_alt_names = vec![SanType::DnsName(format!("{}.device.superinstance.ai", device_id))];
132
133    // Extended key usage for client auth
134    params.extended_key_usages = vec![
135        rcgen::ExtendedKeyUsagePurpose::ClientAuth,
136    ];
137
138    let cert = RcgenCert::from_params(params)
139        .map_err(|e| CloudError::certificate(format!("Failed to generate certificate: {}", e)))?;
140
141    // Convert to rustls types
142    let cert_der = cert.serialize_der()
143        .map_err(|e| CloudError::certificate(format!("Failed to serialize certificate: {}", e)))?;
144    let key_der = cert.serialize_private_key_der();
145
146    Ok((Certificate(cert_der), PrivateKey(key_der)))
147}
