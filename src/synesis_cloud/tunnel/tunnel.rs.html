12/// Main cloud tunnel struct
13///
14/// Provides a persistent QUIC tunnel with mTLS authentication,
15/// automatic heartbeat, and reconnection logic.
16pub struct CloudTunnel {
17    config: TunnelConfig,
18    endpoint: Option<quinn::Endpoint>,
19    connection: Arc<RwLock<Option<quinn::Connection>>>,
20    state_machine: ConnectionStateMachine,
21    heartbeat_service: Option<HeartbeatService>,
22    stats: Arc<RwLock<TunnelStats>>,
23}
24
25impl CloudTunnel {
26    /// Create a new tunnel instance
27    ///
28    /// # Arguments
29    /// * `config` - Tunnel configuration
30    ///
31    /// # Returns
32    /// * Configured tunnel (not yet connected)
33    pub fn new(config: TunnelConfig) -> CloudResult<Self> {
34        if config.cert_path.as_path().as_os_str().is_empty() {
35            return Err(CloudError::validation("Certificate path is required"));
36        }
37        if config.key_path.as_path().as_os_str().is_empty() {
38            return Err(CloudError::validation("Key path is required"));
39        }
40
41        Ok(Self {
42            config,
43            endpoint: None,
44            connection: Arc::new(RwLock::new(None)),
45            state_machine: ConnectionStateMachine::new(),
46            heartbeat_service: None,
47            stats: Arc::new(RwLock::new(TunnelStats::default())),
48        })
49    }
50
51    /// Connect to cloud
52    ///
53    /// # Arguments
54    /// * `cloud_url` - Override cloud URL (optional)
55    ///
56    /// # Returns
57    /// * Ok(()) if connection successful
58    /// * Err if connection fails
59    pub async fn connect(&mut self) -> CloudResult<()> {
60        self.state_machine.transition(TunnelState::Connecting {
61            since: Instant::now(),
62        });
63
64        match self.connect_internal().await {
65            Ok(()) => {
66                // Start heartbeat
67                if let Some(ref heartbeat_service) = self.heartbeat_service {
68                    heartbeat_service.spawn();
69                }
70
71                // Start reconnection monitor
72                // TODO: Implement in next iteration
73
74                tracing::info!("Tunnel connected successfully");
75                Ok(())
76            }
77            Err(e) => {
78                self.state_machine.transition(TunnelState::Failed {
79                    error: e.to_string(),
80                    at: Instant::now(),
81                });
82                Err(e)
83            }
84        }
85    }
86
87    /// Internal connection logic
88    async fn connect_internal(&mut self) -> CloudResult<()> {
89        // Create endpoint if not exists
90        if self.endpoint.is_none() {
91            self.endpoint = Some(create_endpoint(
92                &self.config.cert_path,
93                &self.config.key_path,
94            )?);
95        }
96
97        // Connect to cloud
98        let cloud_url = self.config.cloud_url.clone();
99        let server_name = extract_server_name(&cloud_url)?;
100
101        let conn = connect_to_cloud(
102            self.endpoint.as_ref().unwrap(),
103            &cloud_url,
104            &server_name,
105        ).await?;
106
107        // Store connection
108        *self.connection.write().await = Some(conn.clone());
109
110        // Create and start heartbeat service
111        let heartbeat_service = HeartbeatService::new(HeartbeatConfig::default());
112        heartbeat_service.set_connection(conn.clone()).await;
113        self.heartbeat_service = Some(heartbeat_service);
114
115        // Transition to connected
116        self.state_machine.transition(TunnelState::Connected {
117            since: Instant::now(),
118            latency_ms: 0, // Will be updated by first heartbeat
119        });
120
121        Ok(())
122    }
123
124    /// Disconnect from cloud
125    pub async fn disconnect(&mut self) -> CloudResult<()> {
126        if let Some(ref heartbeat_service) = self.heartbeat_service {
127            heartbeat_service.shutdown();
128            heartbeat_service.clear_connection().await;
129        }
130
131        if let Some(ref conn) = self.connection.write().await.take() {
132            conn.close(0u32.into(), b"client disconnect");
133        }
134
135        self.state_machine.transition(TunnelState::Disconnected);
136
137        tracing::info!("Tunnel disconnected");
138        Ok(())
139    }
140
141    /// Check if connected
142    pub fn is_connected(&self) -> bool {
143        self.state_machine.current().is_connected()
144    }
145
146    /// Get current state
147    pub fn state(&self) -> TunnelState {
148        self.state_machine.current()
149    }
150
151    /// Get tunnel statistics
152    pub async fn stats(&self) -> TunnelStats {
153        self.stats.read().await.clone()
154    }
155
156    /// Send request and receive response (bidirectional stream)
157    pub async fn request(&self, data: &[u8]) -> CloudResult<Vec<u8>> {
158        let conn = self.connection.read().await
159            .as_ref()
160            .ok_or(CloudError::NotConnected)?
161            .clone();
162
163        let (mut send, mut recv) = conn.open_bi().await
164            .map_err(|e| CloudError::tunnel_connection(format!("Failed to open stream: {}", e)))?;
165
166        // Send request
167        send.write_all(data).await
168            .map_err(|e| CloudError::tunnel_connection(format!("Failed to write: {}", e)))?;
169        send.finish().await
170            .map_err(|e| CloudError::tunnel_connection(format!("Failed to finish: {}", e)))?;
171
172        // Receive response
173        let response = recv.read_to_end(10 * 1024 * 1024).await // 10MB max
174            .map_err(|e| CloudError::tunnel_connection(format!("Failed to read: {}", e)))?;
175
176        // Update stats
177        let mut stats = self.stats.write().await;
178        stats.total_bytes_sent += data.len() as u64;
179        stats.total_bytes_received += response.len() as u64;
180        stats.requests_sent += 1;
181        stats.requests_succeeded += 1;
182
183        Ok(response)
184    }
185}

187/// Extract server name from URL
188fn extract_server_name(url: &str) -> CloudResult<String> {
189    let parsed = url::Url::parse(url)
190        .map_err(|e| CloudError::validation(format!("Invalid URL: {}", e)))?;
191
192    parsed.host_str()
193        .map(|s| s.to_string())
194        .ok_or_else(|| CloudError::validation("No host in URL"))
195}
