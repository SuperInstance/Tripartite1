1//! Escalation context builder
2//!
3//! Builder pattern for constructing EscalationContext
4
5use crate::escalation::types::{
6    EscalationContext, KnowledgeChunk, Message, UserPreferences,
7    Verbosity, Tone,
8};
9
10/// Builder for EscalationContext
11pub struct EscalationContextBuilder {
12    context: EscalationContext,
13}
14
15impl Default for EscalationContextBuilder {
16    fn default() -> Self {
17        Self::new()
18    }
19}
20
21impl EscalationContextBuilder {
22    /// Create a new context builder
23    pub fn new() -> Self {
24        Self {
25            context: EscalationContext::default(),
26        }
27    }
28
29    /// Add Pathos intent framing
30    pub fn pathos_framing(mut self, framing: impl Into<String>) -> Self {
31        self.context.pathos_framing = Some(framing.into());
32        self
33    }
34
35    /// Add local knowledge chunk
36    pub fn add_knowledge(mut self, chunk: KnowledgeChunk) -> Self {
37        self.context.local_knowledge.push(chunk);
38        self
39    }
40
41    /// Add multiple knowledge chunks
42    pub fn add_knowledge_chunks(mut self, chunks: Vec<KnowledgeChunk>) -> Self {
43        self.context.local_knowledge.extend(chunks);
44        self
45    }
46
47    /// Add conversation message
48    pub fn add_message(mut self, role: impl Into<String>, content: impl Into<String>) -> Self {
49        self.context.conversation_history.push(Message {
50            role: role.into(),
51            content: content.into(),
52            timestamp: Some(chrono::Utc::now()),
53        });
54        self
55    }
56
57    /// Add user message
58    pub fn user(self, content: impl Into<String>) -> Self {
59        self.add_message("user", content)
60    }
61
62    /// Add assistant message
63    pub fn assistant(self, content: impl Into<String>) -> Self {
64        self.add_message("assistant", content)
65    }
66
67    /// Add constraint
68    pub fn add_constraint(mut self, constraint: impl Into<String>) -> Self {
69        self.context.constraints.push(constraint.into());
70        self
71    }
72
73    /// Add multiple constraints
74    pub fn add_constraints(mut self, constraints: Vec<String>) -> Self {
75        self.context.constraints.extend(constraints);
76        self
77    }
78
79    /// Set user preferences
80    pub fn user_preferences(mut self, prefs: UserPreferences) -> Self {
81        self.context.user_preferences = Some(prefs);
82        self
83    }
84
85    /// Set preferred language
86    pub fn language(mut self, lang: impl Into<String>) -> Self {
87        let mut prefs = self.context.user_preferences.unwrap_or_default();
88        prefs.preferred_language = Some(lang.into());
89        self.context.user_preferences = Some(prefs);
90        self
91    }
92
93    /// Set verbosity
94    pub fn verbosity(mut self, verbosity: Verbosity) -> Self {
95        let mut prefs = self.context.user_preferences.unwrap_or_default();
96        prefs.verbosity = Some(verbosity);
97        self.context.user_preferences = Some(prefs);
98        self
99    }
100
101    /// Set tone
102    pub fn tone(mut self, tone: Tone) -> Self {
103        let mut prefs = self.context.user_preferences.unwrap_or_default();
104        prefs.tone = Some(tone);
105        self.context.user_preferences = Some(prefs);
106        self
107    }
108
109    /// Build the context
110    pub fn build(self) -> EscalationContext {
111        self.context
112    }
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118
119    #[test]
120    fn test_builder_basic() {
121        let context = EscalationContextBuilder::new()
122            .pathos_framing("User wants code help")
123            .add_message("user", "Help me write Rust")
124            .add_constraint("Use safe Rust")
125            .build();
126
127        assert_eq!(context.pathos_framing, Some("User wants code help".to_string()));
128        assert_eq!(context.conversation_history.len(), 1);
129        assert_eq!(context.constraints.len(), 1);
130    }
131
132    #[test]
133    fn test_builder_with_knowledge() {
134        let chunk = KnowledgeChunk {
135            source: "doc.txt".to_string(),
136            content: "Rust is safe".to_string(),
137            relevance: 0.95,
138        };
139
140        let context = EscalationContextBuilder::new()
141            .add_knowledge(chunk)
142            .build();
143
144        assert_eq!(context.local_knowledge.len(), 1);
145        assert_eq!(context.local_knowledge[0].source, "doc.txt");
146    }
147
148    #[test]
149    fn test_builder_with_preferences() {
150        let context = EscalationContextBuilder::new()
151            .language("en")
152            .verbosity(Verbosity::Detailed)
153            .tone(Tone::Technical)
154            .build();
155
156        assert!(context.user_preferences.is_some());
157        let prefs = context.user_preferences.unwrap();
158        assert_eq!(prefs.preferred_language, Some("en".to_string()));
159        assert_eq!(prefs.verbosity, Some(Verbosity::Detailed));
160        assert_eq!(prefs.tone, Some(Tone::Technical));
161    }
162
163    #[test]
164    fn test_builder_convenience_methods() {
165        let context = EscalationContextBuilder::new()
166            .user("What is Rust?")
167            .assistant("Rust is a systems language")
168            .user("Is it safe?")
169            .build();
170
171        assert_eq!(context.conversation_history.len(), 3);
172        assert_eq!(context.conversation_history[0].role, "user");
173        assert_eq!(context.conversation_history[1].role, "assistant");
174        assert_eq!(context.conversation_history[2].role, "user");
175    }
176
177    #[test]
178    fn test_builder_add_constraints_batch() {
179        let constraints = vec![
180            "Use async/await".to_string(),
181            "Handle errors properly".to_string(),
182        ];
183
184        let context = EscalationContextBuilder::new()
185            .add_constraints(constraints)
186            .build();
187
188        assert_eq!(context.constraints.len(), 2);
189    }
